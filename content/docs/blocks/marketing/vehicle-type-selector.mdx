## Vehicle Type Selector

A responsive grid-based vehicle type selector component designed for automotive websites. Features vehicle images, titles, descriptions, and interactive selection states.

## Features

- **Responsive Grid Layout** - 4 columns on desktop, 2 on tablet, 1 on mobile
- **Interactive Selection** - Click to select with visual feedback
- **Highlighted States** - Support for featured/highlighted vehicles
- **Hover Effects** - Smooth scale and shadow transitions
- **Fully Customizable** - Pass custom vehicle data and styling
- **TypeScript Support** - Full type safety with interfaces
- **Accessible** - Semantic HTML with proper button elements

## Usage

### Basic Usage (with default stub data)

```tsx
import { VehicleTypeSelector } from "@/components/blocks/vehicle-type-selector";

export default function VehiclePage() {
  return (
```

### Fetching Data from API

```tsx
"use client";

import { useEffect, useState } from "react";
import { VehicleTypeSelector } from "@/components/blocks/vehicle-type-selector";
import type { VehicleType } from "@/components/blocks/vehicle-type-selector";

export default function DynamicVehicles() {
  const [vehicles, setVehicles] = useState<VehicleType[]>([]);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    async function fetchVehicles() {
      try {
        const response = await fetch("/api/vehicles");
        const data = await response.json();
        setVehicles(data);
      } catch (error) {
        console.error("Failed to fetch vehicles:", error);
      } finally {
        setLoading(false);
      }
    }

    fetchVehicles();
  }, []);

  if (loading) {
    return <div>Loading vehicles...</div>;
  }

  return (
    <VehicleTypeSelector 
      vehicles={vehicles}
      onSelect={(id) => console.log("Selected:", id)}
    />
  );
}
```

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `title` | `string` | `"What type of vehicle?"` | Heading text above the grid |
| `vehicles` | `VehicleType[]` | Default 8 vehicles | Array of vehicle objects (uses stub data if not provided) |
| `onSelect` | `(vehicleId: string) => void` | `undefined` | Callback function when a vehicle is selected |
| `selectedId` | `string` | `undefined` | ID of the currently selected vehicle (for controlled state) |
| `className` | `string` | `undefined` | Additional CSS classes for the container |

### VehicleType Interface

```typescript
export interface VehicleType {
  id: string;              // Unique identifier (required)
  name: string;            // Display name (e.g., "SEDAN", "Camry")
  description: string;     // Short description text
  image: string;           // Image path or URL
  highlighted?: boolean;   // Optional: true = white bg, false/undefined = gray bg
}
```

### VehicleTypeSelectorProps Interface

```typescript
export interface VehicleTypeSelectorProps {
  title?: string;
  vehicles?: VehicleType[];
  onSelect?: (vehicleId: string) => void;
  selectedId?: string;
  className?: string;
}
```

Both interfaces are exported from the component for use in your application.

## Examples

### E-commerce Vehicle Selection

```tsx
"use client";

import { useState } from "react";
import { VehicleTypeSelector } from "@/components/blocks/vehicle-type-selector";
import { Button } from "@/components/ui/button";

export default function VehicleShop() {
  const [selected, setSelected] = useState<string>();

  const handleContinue = () => {
    if (selected) {
      // Navigate to vehicle details
      window.location.href = `/vehicles/${selected}`;
    }
  };

  return (
    <div className="container py-12">
      <VehicleTypeSelector 
        selectedId={selected}
        onSelect={setSelected}
      />
      
      <div className="mt-8 text-center">
        <Button 
          size="lg"
          disabled={!selected}
          onClick={handleContinue}
        >
          Continue with {selected?.toUpperCase()}
        </Button>
      </div>
    </div>
  );
}
```
