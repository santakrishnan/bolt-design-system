## Usage

```tsx

<Toggle variant="outline">
  <Bold className="size-4" />
</Toggle>

// With text
<Toggle variant="outline">
  <Italic className="size-4" />
  <span>Italic</span>
</Toggle>

// Disabled outline toggle
<Toggle variant="outline" disabled>
  <Underline className="size-4" />
  <span>Disabled</span>
</Toggle>
```
