```tsx
import LoadingScreen from "@/components/ui/8bit/blocks/loading-screen";
```

```tsx
<LoadingScreen progress={65} />
```

## Examples

### Auto Progress

Use `autoProgress` for an animated loading bar that automatically progresses from 0 to 100:

```tsx
<LoadingScreen autoProgress autoProgressDuration={5000} />
```

### Custom Tips

Provide your own array of tips to display:

```tsx
<LoadingScreen
  autoProgress
  tips={[
    "Gathering resources...",
    "Building world...",
    "Spawning creatures...",
  ]}
/>
```

### Without Percentage

Hide the percentage display:

```tsx
<LoadingScreen progress={80} showPercentage={false} />
```

### Custom Title

Change the loading title:

```tsx
<LoadingScreen title="PLEASE WAIT..." progress={50} />
```

### Fullscreen Mode

Use the `fullscreen` variant for a full-page loading overlay:

```tsx
<LoadingScreen variant="fullscreen" autoProgress />
```
